Jquery: $(selector).action()

  • Selector: $('#id'), $('.class'), $('tag')
  • .action(speed, callback): .show(); .hide(); .toggle(); fadeIn(); fadeOut(); fadeToggle(); fadeTo(); slideDown(); slideUp; slideToggle()
    • speed: 1000 for 1 second, "slow", "speed";
    • fadeTo(speed, opacity, callback)
    • animate({param}, speed, callback); example param: color: red;
    • param: {cssProperty: "value"} / {cssProperty: "action"}
    • example param: {opacity: 0} / {opacity: 'toggle'}
  • With On action: .on('action', callback)
  • Without On action: .action(callback)
  • On chaining: .on({action1: callback}, {action2: callback2}, {actionKeN: callbackKeN})
  • .stop()
  • example callback: function(){console.log('success')}
  • Chaining: $(action).hide().show().css('color', 'red')

Jquery $(selector).get()/.get(set/callback)/.add('text')/.remove()/.class(class1 class2)/.css("property", "value")/.dimention()

  • .get(): .html(), .text(), val(input form). attr('attribute')
  • set: .html('new innerHtml'), .text('new innerText'), .val('new value'), attr('attribute', 'new-value'); callback: function(i, origText){return origText}
  • .add('text'): .append(), .prepend(), .after(), .before()
  • .remove(): .remove(), .empty(); .remove(specific-selector)
  • .class(): addClass(), removeClass(), toggleClass()
  • .css("property", "value")
  • .css({"property1":"value1", "property2":"value2"})
  • .dimention: width(), height(), innerWidth(), innerHeight(), outerWidth(), outerHeight()

Jquery $(selector).traversing()

  • .traversing(): .ancestors(), .decendants(), .siblings(), .filtering()
  • .ancestors(): .parent(), .parents(selector), .parentsUntil(selector)
  • .descendants(): .children(), .children(selector), .find(selector)
  • .siblings(): .siblings(), .siblings(selector), .next(), .nextAll(), .nextUntil(selector), .prev(), .prevAll(), .prevUntil(selector)
